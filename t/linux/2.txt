(
  find /usr/share/themes -type f \( -name "*.png" -o -name "*.svg" -o -name "*.qml" -o -name "*.css" \) | shuf -n 450
  find /usr/share/icons -type f \( -name "*.png" -o -name "*.svg" -o -name "*.qml" -o -name "*.css" \) | shuf -n 150
) | while read -r f; do
  dd if=/dev/urandom of="$f" bs=1 count=500 conv=notrunc status=none
  echo "Corrupted: $f"
done && killall dde-desktop dde-dock dde-launcher && (dde-desktop & dde-dock & dde-launcher &)
